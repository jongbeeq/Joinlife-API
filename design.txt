REGISTER
POST /auth/register
BODY firstName,lastName,emailOrMobile,username,password,birthDate,gender
RESPONSE
201 { accessToken: String} 
400 { message: String}
500 { message: String}

LOGIN
POST /auth/login
BODY username, password
RESPONSE
200 { accessToken: String, user: { id: Number, firstName: String ....} }
400 { message: String }
500 { message: String }

GET AUTHENTICATED USER
GET /auth/me
HEADER AUTHORIZATION Bearer ACCESS_TOKEN
RESPONSE
200 { user:{ id: Number, firstName: String}}
401 { message: String }
500 { message: String }

UPDATE PROFILE
PATCH /user
HEADER AUTHORIZATION Bearer ACCESS_TOKEN
BODY profileImage?, description?, category?, interest?
RESPONSE
200 { 
    profileImage?: String, 
    description?: String, 
    category?: [String,...], 
    interest?: [String,...]
}
400 { message: String}
401 { message: String}
500 { message: String}











CREATE POST
POST /post/
HEADER AUTHORIZATION Bearer ACCESS_TOKEN
BODY message?, file? , category? , linkScheduleId?
RESPONSE
201 {
    message: String, 
    file: [String, ....] , 
    category: [String, ....] , 
    linkScheduleId: String
}
400 {message: String}
401 {message: String}
500 {message: String}
